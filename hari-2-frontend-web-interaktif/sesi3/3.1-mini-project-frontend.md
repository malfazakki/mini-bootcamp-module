## âš™ï¸ 3. script.js
```javascript
const tombolTambah = document.getElementById("tombol-tambah");
const inputTugas = document.getElementById("input-tugas");
const daftarTugas = document.getElementById("daftar-tugas");

tombolTambah.addEventListener("click", function() {
    const isiTugas = inputTugas.value;
    
    if (isiTugas === "") {
        alert("Isi tugasnya dulu, ya!");
        return;
    }

    // 1. Buat kotak list baru (li)
    const itemBaru = document.createElement("li");
    itemBaru.innerHTML = `
        <span>${isiTugas}</span>
        <button class="btn-hapus">Hapus</button>
    `;

    // 2. Pasang fungsi hapus pada tombol hapus di dalam li tersebut
    itemBaru.querySelector(".btn-hapus").addEventListener("click", function() {
        itemBaru.remove();
    });

    // 3. Masukkan li ke dalam daftar (ul)
    daftarTugas.appendChild(itemBaru);

    // 4. Kosongkan kotak input kembali
    inputTugas.value = "";
});
```
